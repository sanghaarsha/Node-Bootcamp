// 'db' shows all database
> db
natours-test

// collection.insertOne : inerts a document inside a collection

{
        "acknowledged" : true,
        "insertedId" : ObjectId("60fa552d6e973289f554a390")
}

> db 
natours-test

> db.tours.find()
{ "_id" : ObjectId("60fa552d6e973289f554a390"), "name" : "The Forest Hiker", "price" : 297, "rating" : 4.7 }
